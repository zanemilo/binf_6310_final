#!/usr/bin/env bash
#SBATCH --job-name=kallisto_pipeline
#SBATCH --partition=courses
#SBATCH --nodes=1
#SBATCH --ntasks=1
#SBATCH --cpus-per-task=46
#SBATCH --mem=128G
#SBATCH --time=20:00:00
#SBATCH --output=/scratch/polyak.i/ildiko_pipeline_automatic/logs/kallisto_pipeline_%j.log

export PATH=/home/polyak.i/micromamba/envs/binf6310/bin:$PATH

set -euo pipefail

SCRATCH_DIR=/scratch/polyak.i/ildiko_pipeline_automatic
SRA_CACHE_DIR="${SCRATCH_DIR}/results/sra_cache"
KALLISTO_OUT="${SCRATCH_DIR}/results/kallisto"
LOGDIR="${SCRATCH_DIR}/logs"
INDEX="${SCRATCH_DIR}/reference/kallisto/Sus_scrofa.Sscrofa11.1.cdna.all.index"
THREADS=46
SAMPLES_TSV="${1:-}"

if [[ -z "$SAMPLES_TSV" ]]; then
    echo "ERROR: No samples TSV file provided."
    echo "Usage: sbatch kallisto_pipeline.sh all_samples.tsv"
    exit 1
fi

mkdir -p "$SRA_CACHE_DIR" "$KALLISTO_OUT" "$LOGDIR"

echo "Starting Kallisto pipeline..."

while IFS=$'\t' read -r sample_id run_accession genotype region replicate layout; do

    [[ "$sample_id" == "sample_id" ]] && continue

    echo ""
    echo "Processing: $sample_id ($run_accession)"

    SAMPLE_OUT="${KALLISTO_OUT}/${sample_id}"

    if [[ -f "${SAMPLE_OUT}/abundance.tsv" ]]; then
        echo "[skip] Kallisto output already exists for $sample_id"
        continue
    fi

    mkdir -p "$SAMPLE_OUT"

    # Step 1: Download
    echo "[step 1] Downloading $run_accession..."
    prefetch "$run_accession" \
        --output-directory "$SRA_CACHE_DIR" \
        2>&1 | tee "$LOGDIR/prefetch_kallisto_${sample_id}.log"

    # Step 2: Convert to FASTQ
    echo "[step 2] Converting to FASTQ..."
    fasterq-dump "${SRA_CACHE_DIR}/${run_accession}" \
        --split-files \
        --threads "$THREADS" \
        --outdir "$SAMPLE_OUT" \
        --temp /tmp \
        2>&1 | tee "$LOGDIR/fasterq_kallisto_${sample_id}.log"

    # Step 3: Run Kallisto
    echo "[step 3] Running Kallisto..."
    kallisto quant \
        -i "$INDEX" \
        -o "$SAMPLE_OUT" \
        -t "$THREADS" \
        "${SAMPLE_OUT}/${run_accession}_1.fastq" \
        "${SAMPLE_OUT}/${run_accession}_2.fastq" \
        2>&1 | tee "$LOGDIR/kallisto_${sample_id}.log"
    echo "DONE - Kallisto complete for $sample_id"

    # Step 4: Cleanup
    echo "[step 4] Cleaning up..."
    rm -f "${SAMPLE_OUT}/${run_accession}_1.fastq"
    rm -f "${SAMPLE_OUT}/${run_accession}_2.fastq"
    rm -rf "${SRA_CACHE_DIR}/${run_accession}"
    echo "DONE - Cleanup complete for $sample_id"

    echo "COMPLETED: $sample_id"

done < "$SAMPLES_TSV"

echo ""
echo "All samples complete."
echo "Kallisto output: $KALLISTO_OUT"



#Submit the job

#rate = 17,589,296 / 23,585,202 x 100 = 74.6%
#Kallisto's pseudoalignment rate of 74.6% is lower than STAR's alignment rate of 79.49% on the raw unfiltered reads.  This makes sense because Kallisto aligns against the transcriptome only, so rRNA reads that don't match any transcript fail to pseudoalign.

for f in /scratch/polyak.i/ildiko_pipeline_automatic/logs/kallisto_*.log; do
    sample=$(basename $f .log | sed 's/kallisto_//')
    rate=$(grep "reads pseudoaligned" $f | awk '{print $4, $5, $6, $7, $8}')
    echo "$sample: $rate"
done


#check total processed reads to calculate percentages
for f in /scratch/polyak.i/ildiko_pipeline_automatic/logs/kallisto_*.log; do
    sample=$(basename $f .log | sed 's/kallisto_//')
    line=$(grep "reads pseudoaligned" $f)
    processed=$(echo $line | awk '{print $3}' | tr -d ',')
    aligned=$(echo $line | awk '{print $5}' | tr -d ',')
    if [[ -n "$processed" && "$processed" -gt 0 ]]; then
        pct=$(echo "scale=1; $aligned * 100 / $processed" | bc)
        echo "$sample: $pct%"
    fi
done
